```bash
has_glm_zai() {
    # Check for credentials in OpenCode config or env
    [[ -f ~/.local/share/opencode/auth.json ]] && \
        jq -e '.zai.key' ~/.local/share/opencode/auth.json &>/dev/null && return 0
    [[ -f ~/.config/opencode/opencode.json ]] && \
        jq -e '.provider["zai-coding-plan"].options.apiKey' ~/.config/opencode/opencode.json &>/dev/null && return 0
    [[ -n "${ZAI_API_KEY:-}" ]] && return 0
    return 1
}
```

2. **Add run functions** (after existing `run_codex`):
```bash
run_glm_ollama() {
    local prompt="$1"
    local output_file="$2"

    if ! has_glm_ollama; then
        log "Ollama GLM not available"
        return 127
    fi

    log "Starting Ollama GLM task..."
    local exit_code=0
    timeout "$AI_TIMEOUT" ~/dev/.meta/bin/ollama-run "$prompt" "$output_file" || exit_code=$?

    if [[ $exit_code -eq 0 ]]; then
        log "Ollama GLM completed successfully"
    fi
    return $exit_code
}

run_glm_zai() {
    local prompt="$1"
    local output_file="$2"

    if ! has_glm_zai; then
        log "z.ai credentials not found"
        return 127
    fi

    log "Starting z.ai GLM task..."
    local exit_code=0
    timeout "$AI_TIMEOUT" ~/dev/.meta/bin/zai-run "$prompt" "$output_file" || exit_code=$?

    if [[ $exit_code -eq 0 ]]; then
        log "z.ai GLM completed successfully"
    elif [[ $exit_code -eq 2 ]]; then
        log "z.ai rate limit hit"
    fi
    return $exit_code
}
```

| script -> z.ai API | Sending prompts to external cloud API |
| script -> Ollama | Sending prompts to local service |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Information Disclosure | zai-run credential extraction | mitigate | Only extract key, never log it; key stays in memory only |
| T-04-02 | Information Disclosure | prompt content to APIs | accept | User accepts prompts are sent to cloud APIs (intended use) |
| T-04-03 | Tampering | Ollama response injection | accept | Local service, user trusts their own Ollama instance |
| T-04-04 | Denial of Service | Rate limiting | mitigate | Exit code 2 signals retry-able error; ai-delegate fallback chain handles |
