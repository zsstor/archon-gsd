    if ! command -v python3 &>/dev/null; then
        echo "60"
        return
    fi

    local result
    result=$(ISO_TIMESTAMP="$iso_timestamp" python3 <<'EOF' 2>/dev/null
import os
from datetime import datetime, timezone

iso_ts = os.environ['ISO_TIMESTAMP']

try:
    # Handle various ISO8601 formats
    # Remove trailing Z and replace with +00:00
    ts = iso_ts.replace('Z', '+00:00')

    # Handle missing timezone - assume UTC
    if '+' not in ts and '-' not in ts[10:]:
        ts = ts + '+00:00'

    target = datetime.fromisoformat(ts)
    now = datetime.now(timezone.utc)

    # Ensure target is timezone-aware
    if target.tzinfo is None:
        target = target.replace(tzinfo=timezone.utc)

    diff = (target - now).total_seconds()

    # Return positive integer, minimum 1 second
    result = max(1, int(diff))
    print(result)
except Exception:
    print(60)
EOF
    ) || result=""

    # WR-03 fix: validate result is numeric before returning
    [[ "$result" =~ ^[0-9]+$ ]] && echo "$result" || echo "60"
}

local MAX_PARALLEL="${MAX_PARALLEL:-4}"
local running=0

    # Wait if at max capacity
    while [[ $running -ge $MAX_PARALLEL ]]; do
        wait -n 2>/dev/null || true
        ((running--)) || true
    done

    local task_type
    if command -v jq &>/dev/null; then
        task_type=$(echo "$task" | jq -r '.type')
        mapfile -t args < <(echo "$task" | jq -r '.args[]')

        task_type=$(echo "$task" | python3 -c "import json,sys; print(json.load(sys.stdin).get('type',''))")
        mapfile -t args < <(echo "$task" | python3 -c "import json,sys; [print(a) for a in json.load(sys.stdin).get('args',[])]")
    fi

    (
        case "$task_type" in
            impl) cmd_impl "${args[@]}" ;;
            test-pass) cmd_test_pass "${args[@]}" ;;
            scaffold) cmd_scaffold "${args[@]}" ;;
            review) cmd_review "${args[@]}" ;;

        esac
    ) &
    pids+=($!)
