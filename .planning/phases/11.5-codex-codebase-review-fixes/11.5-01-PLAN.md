       local task_type="$1"

           flock -x 9 || { echo "[routing] Warning: Could not acquire config lock" >&2; return 1; }

   import json
   import tempfile
   import shutil
   import os
   import sys

   config_file = os.environ['CONFIG_FILE']
   task_type = os.environ['TASK_TYPE']
   new_mode = os.environ['NEW_MODE']

   try:
       with open(config_file, 'r') as f:
           cfg = json.load(f)

       if 'task_routing' not in cfg:
           cfg['task_routing'] = {}

       task_routing = cfg['task_routing'].get(task_type)
       if isinstance(task_routing, list):

           cfg['task_routing'][task_type] = {
               'models': task_routing,
               'scoring_mode': new_mode

           task_routing['scoring_mode'] = new_mode
       else:
           cfg['task_routing'][task_type] = {
               'scoring_mode': new_mode
           }

       # Atomic write via temp file
       config_dir = os.path.dirname(config_file) or '.'
       fd, temp_path = tempfile.mkstemp(dir=config_dir, prefix='.config-', suffix='.json')
       try:
           with os.fdopen(fd, 'w') as f:
               json.dump(cfg, f, indent=2)
           shutil.move(temp_path, config_file)
       except:
           os.unlink(temp_path)
           raise
