cmd_review() {
    local target="${1:-.}"
    local session="$(session_id)"
    local output_file="$OUTPUT_DIR/review-$session.txt"

    log "Running code review on: $target"

    # Route task
    local model complexity_score scoring_mode failure_rate rationale
    local task_type="code-review"  # WR-06 fix: use consistent task type
    scoring_mode=$(_get_scoring_mode "$task_type")
    complexity_score=$(_score_complexity "$task_type" "Code review: $target")
    failure_rate=$(_compute_weighted_failure_rate "$task_type")
    model=$(route_task "$task_type" "Code review: $target")
    rationale="score=$complexity_score, mode=$scoring_mode, history_fail_rate=$failure_rate"

    # Log routing decision
    local model_chain
    model_chain=$(_get_model_chain "$task_type")
    log_routing_decision "$task_type" "Code review: $target" "$complexity_score" "$scoring_mode" "$failure_rate" "$model" "$model_chain" "$rationale"

    # Review uses backend-specific review command
    local prompt="Perform code review on: $target"

    # Execute with escalation
    local start_time exit_code duration_ms escalation_count final_model

    if [[ $exit_code -eq 0 ]]; then
        log "Review output: $output_file"
        cat "$output_file"
    else
        log "Review failed with exit code $exit_code"
    fi

    return $exit_code
}
