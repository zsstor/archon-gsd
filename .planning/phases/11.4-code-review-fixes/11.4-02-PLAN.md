```bash
# Limit file content size for prompt inclusion (WR-08)
# Max 50 lines, 200 chars per line, 10KB total
_limit_file_content() {
    local file="$1"
    local max_lines="${2:-50}"
    local max_line_length="${3:-200}"
    local max_bytes="${4:-10000}"

    if [[ ! -r "$file" ]]; then
        echo "[File not readable: $file]"
        return
    fi

    head -n "$max_lines" "$file" 2>/dev/null | cut -c1-"$max_line_length" | head -c "$max_bytes"
}
```
