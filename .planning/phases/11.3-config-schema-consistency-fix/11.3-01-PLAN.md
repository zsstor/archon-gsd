**Function to add:**
```bash
# Get model chain array for task type
# Handles both array format (legacy/transition) and object format (with .models)
# Args: task_type
# Returns: JSON array string
_get_model_chain() {
    local task_type="$1"

    if [[ ! -f "$CONFIG_FILE" ]]; then
        echo '["gemini-flash"]'
        return
    fi

    local raw
    if command -v jq &>/dev/null; then
        # Extract .models from object, or use direct value if array (legacy)
        raw=$(jq -r "
            .task_routing.\"${task_type}\" |
            if type == \"object\" then .models else . end //
            .task_routing.impl.models //
            [\"gemini-flash\"]
        " "$CONFIG_FILE" 2>/dev/null)
    else
        raw=$(TASK_TYPE="$task_type" CONFIG_FILE="$CONFIG_FILE" python3 <<'EOF' 2>/dev/null
import json
import os

task_type = os.environ['TASK_TYPE']
config_file = os.environ['CONFIG_FILE']

try:
    with open(config_file) as f:
        cfg = json.load(f)
    routing = cfg.get('task_routing', {})
    entry = routing.get(task_type, routing.get('impl', {'models': ['gemini-flash']}))

    if isinstance(entry, dict):
        print(json.dumps(entry.get('models', ['gemini-flash'])))
    elif isinstance(entry, list):
        print(json.dumps(entry))
    else:
        print('["gemini-flash"]')
except:
    print('["gemini-flash"]')
EOF
)
    fi

    echo "${raw:-[\"gemini-flash\"]}"
}
```
