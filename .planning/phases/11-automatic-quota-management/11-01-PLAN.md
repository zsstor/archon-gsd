_get_failover_candidates() {
    local current_model="$1"
    local task_type="$2"

    local current_tier
    current_tier=$(_get_model_tier "$current_model")

import json, sys, os
chain = json.load(sys.stdin)
current_model = os.environ["CURRENT_MODEL"]
current_tier = int(os.environ["CURRENT_TIER"])

TIERS = {

3. `_check_version_allowed()` — Version pinning per Q-05:
   - Args: model, version
   - Read allowed_versions and forbidden_versions from config

```bash
_check_version_allowed() {
    local model="$1"
    local version="$2"

    # Check forbidden list first
    local forbidden
    forbidden=$(read_config "models.${model}.forbidden_versions" "[]")
    if echo "$forbidden" | grep -q "\"$version\""; then
        return 1
    fi

    # Check allowed list (if empty, all allowed)
    local allowed
    allowed=$(read_config "models.${model}.allowed_versions" "[]")
    if [[ "$allowed" == "[]" ]]; then
        return 0  # No restrictions
    fi

    if echo "$allowed" | grep -q "\"$version\""; then
        return 0
    fi

    return 1  # Not in allowed list
}
```

_should_wait_not_failover() {
    local task_type="$1"
    local current_model="$2"
    local wait_seconds="$3"

    # D-11: opus-only always waits

}

```bash
#!/usr/bin/env bash
# gemma-parse - Local LLM wrapper for quota error parsing
# Uses Ollama with GLM-5.1 or Gemma for truly local inference
# Per D-02: Must be local (not subject to API quota itself)

set -euo pipefail

# Default model (GLM via Ollama is available per research verification)
OLLAMA_MODEL="${QUOTA_PARSE_MODEL:-glm-5.1:cloud}"
OLLAMA_TIMEOUT="${OLLAMA_TIMEOUT:-30}"

# Check Ollama is running
if ! curl -s --max-time 2 http://localhost:11434/ 2>/dev/null | grep -qi "ollama"; then
    echo "60"  # Default fallback
    exit 0
fi

# Input: error content (from stdin or file)
error_content=""
if [[ $# -gt 0 && -f "$1" ]]; then
    error_content=$(head -c 2000 "$1")
elif [[ ! -t 0 ]]; then
    error_content=$(head -c 2000)
else
    echo "60"
    exit 0
fi

# Prompt for extracting wait time
prompt="Extract the wait time from this API error message.
Return ONLY a single integer representing seconds to wait.
If the message mentions minutes, convert to seconds.
If no wait time is found, return 60.

Error message:
$error_content"

# Call Ollama
result=$(timeout "$OLLAMA_TIMEOUT" ollama run "$OLLAMA_MODEL" "$prompt" 2>/dev/null | grep -oE '^[0-9]+' | head -1)

# Return result or default
echo "${result:-60}"
```
