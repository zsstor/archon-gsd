```bash
        2)  # Quota error - apply wait-before-failover logic per D-08
            signal="quota"

            # Parse wait time from error response
            local wait_seconds
            wait_seconds=$(_parse_quota_error "$output_file")

            # Decide: wait or failover?
            if _should_wait_not_failover "$task_type" "$model" "$wait_seconds"; then
                # D-08: Sleep in-process, preserve state
                [[ "${VERBOSE:-false}" == "true" ]] && \
                    echo "[quota] $model quota-limited, waiting ${wait_seconds}s for recovery..." >&2

                # Record state so other processes know this model is limited
                local recovery_time
                recovery_time=$(($(date +%s) + wait_seconds))
                _record_quota_state "$model" "$recovery_time"

                # Log quota wait event
                log_quota_event "$task_type" "$model" "wait" "$wait_seconds"

                # Sleep and retry same model (don't escalate yet)
                sleep "$wait_seconds"

                # Add jitter to prevent thundering herd (0-5 seconds)
                sleep $((RANDOM % 6))

                # Retry: decrement i so the loop tries this model again
                ((i--))
                signal=""  # Clear signal so we don't escalate
                continue
            else
                # Wait time too long, proceed to failover
                [[ "${VERBOSE:-false}" == "true" ]] && \
                    echo "[quota] $model quota-limited for ${wait_seconds}s (>${MAX_WAIT_SECONDS}), failing over..." >&2

                # Get tier-aware failover candidates
                local failover_candidates
                failover_candidates=$(_get_failover_candidates "$model" "$task_type")

                if [[ "$failover_candidates" == "[]" ]]; then
                    # No failover candidates (e.g., opus-only task) - must wait
                    [[ "${VERBOSE:-false}" == "true" ]] && \
                        echo "[quota] No failover candidates available, waiting anyway..." >&2
                    _record_quota_state "$model" "$(($(date +%s) + wait_seconds))"
                    log_quota_event "$task_type" "$model" "forced-wait" "$wait_seconds"
                    sleep "$wait_seconds"
                    sleep $((RANDOM % 6))
                    ((i--))
                    signal=""
                    continue
                fi

                # Log failover event
                log_quota_event "$task_type" "$model" "failover" "$wait_seconds"

                # Mark current model as limited so it's skipped in failover selection
                _record_quota_state "$model" "$(($(date +%s) + wait_seconds))"
            fi
            ;;
```

```bash
# ============================================================================
# Quota Event Logging
# ============================================================================

# Log quota event for debugging and analysis
# Args: task_type, model, action (wait|failover|forced-wait), wait_seconds
# Writes to: .planning/escalation-log.jsonl (same as escalation events)
log_quota_event() {
    local task_type="$1"
    local model="$2"
    local action="$3"
    local wait_seconds="$4"

    local escalation_log="${PROJECT_ROOT}/.planning/escalation-log.jsonl"
    mkdir -p "$(dirname "$escalation_log")" 2>/dev/null || true

    # Build JSON entry using Python for reliable serialization
    TASK_TYPE="$task_type" \
    MODEL="$model" \
    ACTION="$action" \
    WAIT_SECONDS="$wait_seconds" \
    python3 <<'EOF' >> "$escalation_log" 2>/dev/null || true
import json
import os
from datetime import datetime

entry = {
    "timestamp": datetime.now().isoformat(),
    "event_type": "quota",
    "task_type": os.environ['TASK_TYPE'],
    "model": os.environ['MODEL'],
    "action": os.environ['ACTION'],
    "wait_seconds": int(os.environ.get('WAIT_SECONDS', '0'))
}
print(json.dumps(entry))
EOF

    if [[ "${VERBOSE:-false}" == "true" ]]; then
        echo "[quota] Logged: model=$model action=$action wait=${wait_seconds}s" >&2
    fi
}

# Query quota events for model (useful for debugging)
# Args: model, max_age_hours (default 24)
# Returns: JSON array of matching entries
query_quota_events() {
    local model="$1"
    local max_age_hours="${2:-24}"

    local escalation_log="${PROJECT_ROOT}/.planning/escalation-log.jsonl"
    if [[ ! -f "$escalation_log" ]]; then
        echo "[]"
        return
    fi

    python3 <<EOF 2>/dev/null || echo "[]"
import json
from datetime import datetime, timedelta

results = []
cutoff = datetime.now() - timedelta(hours=$max_age_hours)

try:
    with open('$escalation_log') as f:
        for line in f:
            try:
                entry = json.loads(line.strip())
                if entry.get('event_type') != 'quota':
                    continue
                if entry.get('model') != '$model':
                    continue

                # Check age
                ts_str = entry.get('timestamp', '')
                try:
                    ts = datetime.fromisoformat(ts_str.replace('Z', '+00:00'))
                    if ts < cutoff:
                        continue
                except:
                    pass

                results.append(entry)
            except:
                continue
except:
    pass

print(json.dumps(results))
EOF
}
```
