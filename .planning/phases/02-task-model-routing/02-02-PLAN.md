   **cmd_impl()** (rewritten):
   ```bash
   cmd_impl() {
       local description="${1:-}"
       local context_file="${2:-}"
       [[ -z "$description" ]] && die "Usage: ai-delegate impl <description> [context_file]"

       local session="$(session_id)"
       local output_file="$OUTPUT_DIR/impl-$session.txt"

       # Route task using routing module
       local model complexity_score scoring_mode failure_rate rationale
       scoring_mode=$(_get_scoring_mode "impl")
       complexity_score=$(_score_complexity "impl" "$description")
       failure_rate=$(_compute_weighted_failure_rate "impl")
       model=$(route_task "impl" "$description" "$context_file")
       rationale="score=$complexity_score, mode=$scoring_mode, history_fail_rate=$failure_rate"

       # Log routing decision (always per D-15)
       local model_chain
       model_chain=$(read_config "task_routing.impl" '["gemini-flash"]')
       log_routing_decision "impl" "$description" "$complexity_score" "$scoring_mode" "$failure_rate" "$model" "$model_chain" "$rationale"

       # Verbose output (per D-13)
       if [[ "$VERBOSE" == "true" ]]; then
           verbose_log "Routing: $rationale"
           verbose_log "Selected: $model"
       fi

       # Build prompt
       local prompt="$description"
       if [[ -n "$context_file" && -f "$PROJECT_ROOT/$context_file" ]]; then
           prompt="$description

   Context from $context_file:
   $(cat "$PROJECT_ROOT/$context_file")

   Implement this following the existing patterns in the codebase."
       fi

       # Execute using execution module
       local start_time exit_code duration_ms
       start_time=$(date +%s%3N)
       execute_model "$model" "$prompt" "$output_file" "$AI_TIMEOUT"
       exit_code=$?
       duration_ms=$(($(date +%s%3N) - start_time))

       # Log outcome
       local success="true"
       [[ $exit_code -ne 0 ]] && success="false"
       log_task_outcome "impl" "$model" "$success" "$duration_ms" "0" "0"

       return $exit_code
   }
   ```
