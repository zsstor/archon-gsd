# Loop detection using Python difflib.SequenceMatcher (per D-01)
# Args: output_file, history_file, threshold (default 0.85)
# Returns: 0 if loop detected, 1 otherwise
_detect_loop() {
    local output_file="$1"
    local history_file="$2"
    local threshold="${3:-0.85}"

    [[ ! -f "$output_file" ]] && return 1
    [[ ! -f "$history_file" ]] && return 1

    python3 <<EOF
import sys
import difflib
import json

try:
    new_output = open('$output_file').read()
    with open('$history_file') as f:
        # Read last 3 outputs from JSONL history
        history = [json.loads(line)['output'] for line in f.readlines()[-3:]]
except:
    sys.exit(1)  # No history or error, no loop

for prev_output in history:
    ratio = difflib.SequenceMatcher(None, new_output, prev_output).ratio()
    if ratio >= $threshold:
        sys.exit(0)  # Loop detected
sys.exit(1)  # No loop
EOF
}

# Test failure detection using regex (per D-02)
# Args: output_file
# Returns: 0 if test failure detected, 1 otherwise
_detect_test_failure() {
    local output_file="$1"
    [[ ! -f "$output_file" ]] && return 1

    # Framework-agnostic patterns: pytest, jest, vitest, playwright, generic
    grep -qiE "(FAIL|FAILED|AssertionError|Error:|FAILED TESTS|Test Suite Failed|✗.*fail|[0-9]+ failed)" "$output_file" 2>/dev/null
}

# Token exhaustion detection (per D-03)
# Args: output_file
# Returns: 0 if token exhaustion detected, 1 otherwise
_detect_token_exhaustion() {
    local output_file="$1"
    [[ ! -f "$output_file" ]] && return 1

    # Provider-agnostic patterns
    grep -qiE "(max_tokens|truncated|context_length|exceed.*context|token.*limit|maximum.*tokens)" "$output_file" 2>/dev/null
}
```

# Build structured markdown handoff context for escalation
# Args: attempt_number, failed_model, output_file, failure_signal
# Returns: markdown string to append to prompt
_build_handoff_context() {
    local attempt_number="$1"
    local failed_model="$2"
    local output_file="$3"
    local failure_signal="$4"

    # Truncate output to first 100 lines to prevent context bloat
    local output_preview=""
    if [[ -f "$output_file" ]]; then
        output_preview=$(head -100 "$output_file")
    fi

    # Generate failure explanation
    local failure_explanation
    case "$failure_signal" in
        loop) failure_explanation="Output repeated similar pattern from previous attempt. The approach is not progressing." ;;
        test_failure) failure_explanation="Implementation produced failing tests. Review the assertion errors below." ;;
        token_exhaustion) failure_explanation="Context length exceeded model capacity. Consider a more focused approach." ;;
        timeout) failure_explanation="Execution exceeded timeout. The task may be too complex or model is stuck." ;;
        quota) failure_explanation="Provider quota/rate limit reached. This is a resource constraint, not a capability issue." ;;
        explicit_failure) failure_explanation="Execution failed with non-zero exit code. Review error output below." ;;
        *) failure_explanation="Unknown failure signal: $failure_signal" ;;
    esac

    cat <<EOF

---

## Previous Attempts

### Attempt ${attempt_number}: ${failed_model}

**Failure Signal:** ${failure_signal}

**What went wrong:** ${failure_explanation}

**Output (truncated to 100 lines):**
\`\`\`
${output_preview}
\`\`\`

**Instructions for next attempt:**
- Review what was tried above
- DO NOT repeat the same approach that failed
- Consider an alternative implementation strategy
- If tests failed, analyze the specific assertion errors and fix the root cause
- If loop detected, try a fundamentally different approach

EOF
}
```

1. Main escalation entry point (per D-07, D-08, D-09):
```bash
# Execute task with automatic escalation on failure
# Args: task_type, initial_model, base_prompt, output_file, timeout
# Returns: 0 on success, 1 on chain exhausted
# Implements: D-07 (immediate escalation), D-08 (fail with history), D-09 (no cooldown)
execute_with_escalation() {
    local task_type="$1"
    local initial_model="$2"
    local base_prompt="$3"
    local output_file="$4"
    local timeout="${5:-${AI_TIMEOUT:-300}}"

    # Get escalation chain from config
    local chain_key="escalation.chain.${task_type}"
    local chain_json
    chain_json=$(read_config "$chain_key" "$(read_config "escalation.chain.default" '[]')")

    # Parse chain into array
    local -a chain
    mapfile -t chain < <(echo "$chain_json" | python3 -c "import json,sys; [print(m) for m in json.load(sys.stdin)]" 2>/dev/null)

    # Empty chain fallback
    if [[ ${#chain[@]} -eq 0 ]]; then
        chain=("gemini-flash" "glm-ollama" "codex" "claude-opus")
    fi

    # Find starting position in chain
    local start_idx=0
    for i in "${!chain[@]}"; do
        [[ "${chain[$i]}" == "$initial_model" ]] && start_idx=$i && break
    done

    # Session tracking
    local session="${SESSION_ID:-$(date +%Y%m%d-%H%M%S)-$$}"
    local history_file="${OUTPUT_DIR:-.ai-output}/history-${session}.jsonl"
    local attempt=1
    local prompt="$base_prompt"
    local accumulated_handoff=""

    # Walk the chain (D-07: immediate escalation on any signal)
    for ((i=start_idx; i<${#chain[@]}; i++)); do
        local model="${chain[$i]}"

        [[ "${VERBOSE:-false}" == "true" ]] && echo "[escalation] Attempt $attempt: trying $model" >&2

        # Execute model (from execution.sh)
        local start_time exit_code
        start_time=$(date +%s%3N)
        execute_model "$model" "$prompt" "$output_file" "$timeout"
        exit_code=$?

        # Detect signals (D-01, D-02, D-03, D-04)
        local signal=""
        case $exit_code in
            0) ;;  # Success - check for other signals
            2) signal="quota" ;;
            124) signal="timeout" ;;
            127) signal="not_available"; ((i--)); continue ;;  # Skip unavailable, try next
            *) signal="explicit_failure" ;;
        esac

        # Additional signal detection if no signal yet
        if [[ -z "$signal" ]]; then
            if _detect_loop "$output_file" "$history_file" 0.85; then
                signal="loop"
            elif _detect_test_failure "$output_file"; then
                signal="test_failure"
            elif _detect_token_exhaustion "$output_file"; then
                signal="token_exhaustion"
            fi
        fi

        # Record output for future loop detection
        _record_output_history "$output_file" "$history_file" "$session"

        # Success path (D-11 will be handled by caller via log_task_outcome)
        if [[ -z "$signal" ]]; then
            [[ "${VERBOSE:-false}" == "true" ]] && echo "[escalation] Success on attempt $attempt ($model)" >&2
            # Return escalation count for logging
            echo "$((attempt - 1))" > "${output_file}.escalation_count"
            return 0
        fi

        [[ "${VERBOSE:-false}" == "true" ]] && echo "[escalation] $model failed: $signal. Escalating..." >&2

        # Build handoff context (D-05, D-06)
        accumulated_handoff+="$(_build_handoff_context "$attempt" "$model" "$output_file" "$signal")"
        prompt="${base_prompt}${accumulated_handoff}"

        ((attempt++))
    done

    # Chain exhausted (D-08)
    [[ "${VERBOSE:-false}" == "true" ]] && echo "[escalation] All models in chain failed after $((attempt-1)) attempts" >&2
    echo "$((attempt - 1))" > "${output_file}.escalation_count"
    return 1
}
```

# Validate escalation chain has no duplicates and ends with capable model
# Args: task_type
# Returns: 0 if valid, 1 if issues found (warnings to stderr)
_validate_escalation_chain() {
    local task_type="$1"

    local chain_json
    chain_json=$(read_config "escalation.chain.${task_type}" "$(read_config "escalation.chain.default" '[]')")

import sys

try:

# Check for duplicates
if len(chain) != len(set(chain)):
    print(f"[warning] escalation chain has duplicates: {chain}", file=sys.stderr)
    sys.exit(1)

# Check chain ends with high-capability model
high_capability = ['claude-opus', 'claude-sonnet', 'codex']
if chain and chain[-1] not in high_capability:
    print(f"[warning] escalation chain should end with {high_capability}, got {chain[-1]}", file=sys.stderr)

sys.exit(0)
EOF
}
