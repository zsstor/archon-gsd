```bash
# Log escalation event to separate verbose log (per D-12)
# Args: task_type, attempt_number, model, failure_signal, output_file
# Writes to: .planning/escalation-log.jsonl
log_escalation_event() {
    local task_type="$1"
    local attempt_number="$2"
    local model="$3"
    local failure_signal="$4"
    local output_file="$5"

    local escalation_log="${PROJECT_ROOT}/.planning/escalation-log.jsonl"
    mkdir -p "$(dirname "$escalation_log")" 2>/dev/null || true

    # Read output preview (first 500 chars for verbose log)
    local output_preview=""
    if [[ -f "$output_file" ]]; then
        output_preview=$(head -c 500 "$output_file" 2>/dev/null || true)
    fi

    # Build JSON entry using Python for reliable serialization
    python3 <<EOF >> "$escalation_log" 2>/dev/null || true
import json
from datetime import datetime

# Escape special characters in output preview
output_preview = '''${output_preview//\'/\\\'}'''[:500]

entry = {
    "timestamp": datetime.now().isoformat(),
    "event_type": "escalation",
    "task_type": """$task_type""",
    "attempt": $attempt_number,
    "model": """$model""",
    "signal": """${failure_signal:-unknown}""",
    "output_preview": output_preview
}
print(json.dumps(entry))
EOF

    if [[ "${VERBOSE:-false}" == "true" ]]; then
        echo "[escalation] Logged: attempt=$attempt_number model=$model signal=$failure_signal" >&2
    fi
}
```

Also add log rotation check for escalation log:

```bash
# Check if escalation log rotation needed (mirrors delegation log rotation)
_should_rotate_escalation_log() {
    local escalation_log="${PROJECT_ROOT}/.planning/escalation-log.jsonl"
    if [[ ! -f "$escalation_log" ]]; then
        return 1
    fi

    local line_count
    line_count=$(wc -l < "$escalation_log" 2>/dev/null || echo "0")

    # Rotate after 10,000 entries (same as delegation log)
    if [[ $line_count -gt 10000 ]]; then
        return 0
    fi

    return 1
}

# Rotate escalation log file
_rotate_escalation_log() {
    local escalation_log="${PROJECT_ROOT}/.planning/escalation-log.jsonl"
    if [[ ! -f "$escalation_log" ]]; then
        return
    fi

    local archive_name="${escalation_log%.jsonl}-$(date +%Y%m).jsonl"
    mv "$escalation_log" "$archive_name" 2>/dev/null || return 1
    gzip -f "$archive_name" 2>/dev/null || true

    if [[ "${VERBOSE:-false}" == "true" ]]; then
        echo "[logging] Rotated escalation log to ${archive_name}.gz" >&2
    fi
}
```
